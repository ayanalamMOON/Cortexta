```ts
const client = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
```

## Step 4: Mempalace (Core Memory Engine)

```ts
// src/services/mempalace.ts
import { embed } from "./embedding";

const memoryStore: any[] = [];

export async function storeMemory(data: any) {
  const embedding = await embed(data.content);

  const memory = {
    ...data,
    embedding,
    createdAt: new Date(),
  };

  memoryStore.push(memory);
  return memory;
}
```

---

## Step 5: Semantic Retrieval

```ts
// src/services/retrieval.ts
import cosineSimilarity from "cosine-similarity";
import { embed } from "./embedding";

export async function searchMemory(query: string) {
  const queryEmbedding = await embed(query);

  return memoryStore
    .map((m) => ({
      ...m,
      score: cosineSimilarity(queryEmbedding, m.embedding),
    }))
    .sort((a, b) => b.score - a.score)
    .slice(0, 5);
}
```

## 1) Tree-sitter based AST extraction

This version does four things:

1. detects language from file extension
2. parses the file with Tree-sitter
3. extracts semantic entities like functions, classes, interfaces, loops, branches
4. converts them into compact memory atoms

```ts
import path from "node:path";
import Parser from "tree-sitter";

const TypeScriptGrammar = require("tree-sitter-typescript");
const CppGrammar = require("tree-sitter-cpp");

export type SupportedLanguage = "typescript" | "tsx" | "cpp";

export interface ParserBundle {
  parser: any;
  language: SupportedLanguage;
}

export function resolveLanguage(filePath: string): SupportedLanguage | null {
  const ext = path.extname(filePath).toLowerCase();

  if (ext === ".ts") return "typescript";
  if (ext === ".tsx") return "tsx";
  if (ext === ".cpp" || ext === ".cc" || ext === ".cxx" || ext === ".hpp" || ext === ".h") return "cpp";

  return null;
}

export function createParserForFile(filePath: string): ParserBundle {
  const language = resolveLanguage(filePath);
  if (!language) {
    throw new Error(`Unsupported file type: ${filePath}`);
  }

  const parser = new Parser();

  if (language === "typescript") {
    parser.setLanguage(TypeScriptGrammar.typescript);
  } else if (language === "tsx") {
    parser.setLanguage(TypeScriptGrammar.tsx);
  } else if (language === "cpp") {
    const cppLanguage = CppGrammar.default ?? CppGrammar;
    parser.setLanguage(cppLanguage);
  }

  return { parser, language };
}
```

```ts
import { safeSnippet } from "../utils/text";

const LOOP_TYPES = new Set(["for_statement", "while_statement", "do_statement", "for_in_statement", "for_of_statement"]);
const BRANCH_TYPES = new Set(["if_statement", "switch_statement", "conditional_expression", "ternary_expression"]);
const FUNC_TYPES = new Set(["function_declaration", "function_definition", "method_definition", "arrow_function"]);
const CLASS_TYPES = new Set(["class_declaration", "struct_specifier"]);
const INTERFACE_TYPES = new Set(["interface_declaration"]);
const MODULE_TYPES = new Set(["program", "translation_unit"]);

export function classifyNodeType(nodeType: string): "function" | "class" | "interface" | "struct" | "method" | "loop" | "branch" | "module" | null {
  if (nodeType === "method_definition") return "method";
  if (LOOP_TYPES.has(nodeType)) return "loop";
  if (BRANCH_TYPES.has(nodeType)) return "branch";
  if (FUNC_TYPES.has(nodeType)) return "function";
  if (INTERFACE_TYPES.has(nodeType)) return "interface";
  if (CLASS_TYPES.has(nodeType)) return nodeType === "struct_specifier" ? "struct" : "class";
  if (MODULE_TYPES.has(nodeType)) return "module";
  return null;
}

export function makeSummary(params: {
  kind: string;
  name: string;
  signature: string;
  complexityHint: string;
  dependencies: string[];
  sourcePreview: string;
}): string {
  const deps = params.dependencies.length ? params.dependencies.slice(0, 8).join(", ") : "none";
  return [
    `${params.kind.toUpperCase()} ${params.name}`,
    `signature: ${params.signature}`,
    `complexity: ${params.complexityHint}`,
    `dependencies: ${deps}`,
    `preview: ${safeSnippet(params.sourcePreview, 180)}`
  ].join("\n");
}
```

### `packages/core/src/ast/extractor.ts`

```ts
import fs from "node:fs";
import path from "node:path";
import { createHash } from "node:crypto";
import { CodeEntity, ExtractedFile } from "../types/memory";
import { createParserForFile, resolveLanguage } from "./language-registry";
import { classifyNodeType, makeSummary } from "./summarize";
import { normalizeWhitespace } from "../utils/text";
import { sha256 } from "../utils/hash";

const NAME_FIELD_CANDIDATES = ["name", "declarator", "identifier", "field_identifier", "type_identifier", "property_identifier"];

const CONTROL_NODE_TYPES = new Set([
  "if_statement",
  "switch_statement",
  "for_statement",
  "while_statement",
  "do_statement",
  "for_in_statement",
  "for_of_statement",
  "conditional_expression"
]);

function sliceNode(source: string, node: any): string {
  return source.slice(node.startIndex, node.endIndex);
}

function firstNamedChildByTypes(node: any, types: string[]): any | null {
  if (!node || !node.namedChildren) return null;
  for (const child of node.namedChildren) {
    if (types.includes(child.type)) return child;
  }
  for (const child of node.namedChildren) {
    const found = firstNamedChildByTypes(child, types);
    if (found) return found;
  }
  return null;
}

function findNameText(node: any, source: string): string {
  for (const field of NAME_FIELD_CANDIDATES) {
    try {
      const n = node.childForFieldName?.(field);
      if (n) return normalizeWhitespace(sliceNode(source, n));
    } catch {
      // ignore
    }
  }

  const maybeNameNode = firstNamedChildByTypes(node, ["identifier", "type_identifier", "field_identifier", "property_identifier"]);
  if (maybeNameNode) {
    return normalizeWhitespace(sliceNode(source, maybeNameNode));
  }

  return "anonymous";
}

function countPatterns(root: any, predicate: (type: string) => boolean): number {
  let count = 0;

  function walk(node: any): void {
    if (predicate(node.type)) count += 1;
    if (node.namedChildren) {
      for (const child of node.namedChildren) walk(child);
    }
  }

  walk(root);
  return count;
}

function collectIdentifiers(root: any, source: string): Set<string> {
  const ids = new Set<string>();

  function walk(node: any): void {
    if (node.type === "identifier" || node.type === "field_identifier" || node.type === "property_identifier" || node.type === "type_identifier") {
      const text = normalizeWhitespace(sliceNode(source, node));
      if (text && text.length > 1) ids.add(text);
    }
    if (node.namedChildren) {
      for (const child of node.namedChildren) walk(child);
    }
  }

  walk(root);
  return ids;
}

function inferComplexityHint(root: any, source: string, functionName: string): string {
  const loopCount = countPatterns(root, (t) => t.includes("for_") || t === "for_statement" || t === "while_statement" || t === "do_statement");
  const branchCount = countPatterns(root, (t) => t === "if_statement" || t === "switch_statement" || t === "conditional_expression");
  const identifiers = collectIdentifiers(root, source);
  const isRecursive = identifiers.has(functionName);

  if (loopCount >= 2) return `likely O(n^${Math.min(loopCount, 3)}) due to nested loops`;
  if (loopCount === 1 && branchCount > 0) return "likely O(n) with branching";
  if (loopCount === 1) return "likely O(n)";
  if (isRecursive) return "recursive, inspect stack growth and termination";
  if (branchCount > 0) return "branch-heavy, data-dependent cost";
  return "structural helper, cost likely linear in its input size";
}

function collectDependencies(root: any, source: string): string[] {
  const deps = new Set<string>();

  function walk(node: any): void {
    if (node.type === "call_expression") {
      const callee = node.childForFieldName?.("function") ?? node.childForFieldName?.("name");
      if (callee) {
        const text = normalizeWhitespace(sliceNode(source, callee));
        if (text && text.length > 1) deps.add(text);
      }
    }

    if (node.type === "import_statement" || node.type === "import_declaration" || node.type === "using_declaration") {
      const text = normalizeWhitespace(sliceNode(source, node));
      if (text) deps.add(text);
    }

    if (node.namedChildren) {
      for (const child of node.namedChildren) walk(child);
    }
  }

  walk(root);
  return [...deps].slice(0, 12);
}

function buildEntity(params: {
  kind: NonNullable<ReturnType<typeof classifyNodeType>>;
  language: string;
  filePath: string;
  node: any;
  source: string;
  root: any;
}): CodeEntity {
  const name = findNameText(params.node, params.source);
  const sourceSlice = sliceNode(params.source, params.node);
  const signature = normalizeWhitespace(sourceSlice.split("{")[0].split("=>")[0]).slice(0, 220);
  const complexityHint = inferComplexityHint(params.node, params.source, name);
  const dependencies = collectDependencies(params.node, params.source);
  const hash = sha256(`${params.filePath}:${params.node.startIndex}:${params.node.endIndex}:${normalizeWhitespace(sourceSlice)}`);

  return {
    id: hash.slice(0, 24),
    kind: params.kind,
    language: params.language,
    filePath: params.filePath,
    name,
    signature,
    summary: makeSummary({
      kind: params.kind,
      name,
      signature,
      complexityHint,
      dependencies,
      sourcePreview: sourceSlice
    }),
    complexityHint,
    dependencies,
    startLine: params.node.startPosition.row + 1,
    endLine: params.node.endPosition.row + 1,
    startIndex: params.node.startIndex,
    endIndex: params.node.endIndex,
    sourceHash: sha256(normalizeWhitespace(sourceSlice)),
    rawSlice: sourceSlice,
    createdAt: Date.now()
  };
}

function visit(node: any, source: string, language: string, filePath: string, entities: CodeEntity[]): void {
  const kind = classifyNodeType(node.type);

  if (kind && node.type !== "program" && node.type !== "translation_unit") {
    if (kind === "loop" || kind === "branch") {
      entities.push(buildEntity({ kind, language, filePath, node, source, root: node }));
    } else {
      entities.push(buildEntity({ kind, language, filePath, node, source, root: node }));
    }
  }

  if (node.namedChildren) {
    for (const child of node.namedChildren) {
      visit(child, source, language, filePath, entities);
    }
  }
}

export function extractEntitiesFromFile(filePath: string): ExtractedFile {
  const language = resolveLanguage(filePath);
  if (!language) {
    return { filePath, language: "unsupported", entities: [] };
  }

  const source = fs.readFileSync(filePath, "utf8");
  const { parser } = createParserForFile(filePath);
  const tree = parser.parse(source);

  const entities: CodeEntity[] = [];
  visit(tree.rootNode, source, language, filePath, entities);

  return {
    filePath,
    language,
    entities: dedupeEntities(entities)
  };
}

function dedupeEntities(entities: CodeEntity[]): CodeEntity[] {
  const seen = new Set<string>();
  const out: CodeEntity[] = [];

  for (const e of entities) {
    const key = `${e.kind}:${e.startIndex}:${e.endIndex}:${e.sourceHash}`;
    if (seen.has(key)) continue;
    seen.add(key);
    out.push(e);
  }

  return out;
}
```

## 2) Exact token counting and context compiler

This compiler uses the final rendered context string as the source of truth. That is the cleanest way to keep the budget exact, because you count the same bytes that will be sent to the model. `tiktoken` is suitable here because it provides a local tokenizer compatible with model encodings. ([GitHub][2])

```ts
import { encoding_for_model, get_encoding } from "tiktoken";

export class TokenCounter {
  private encoding: any;

  constructor(model: string = "gpt-4o-mini") {
    try {
      this.encoding = encoding_for_model(model as any);
    } catch {
      this.encoding = get_encoding("cl100k_base");
    }
  }

  countText(text: string): number {
    return this.encoding.encode(text).length;
  }

  countJson(value: unknown): number {
    return this.countText(JSON.stringify(value));
  }

  free(): void {
    this.encoding?.free?.();
  }
}
```

```ts
import { ContextAtom } from "../types/context";

export function renderAtom(atom: ContextAtom, mode: "full" | "compact" | "symbolic" = "compact"): string {
  const tags = atom.tags?.length ? ` tags=${atom.tags.join(",")}` : "";
  const ref = atom.sourceRef ? ` ref=${atom.sourceRef}` : "";

  if (mode === "full") {
    return [
      `- id: ${atom.id}`,
      `  kind: ${atom.kind}`,
      `  title: ${atom.title}`,
      `  priority: ${atom.priority.toFixed(3)}`,
      `  relevance: ${atom.relevance.toFixed(3)}`,
      `  recency: ${atom.recency.toFixed(3)}`,
      `  body: ${atom.body}`,
      `  ${tags.trim()}`,
      `  ${ref.trim()}`
    ].join("\n");
  }

  if (mode === "symbolic") {
    return `- ${atom.id} | ${atom.kind} | ${atom.title} | p=${atom.priority.toFixed(2)} r=${atom.relevance.toFixed(2)} t=${atom.recency.toFixed(2)} | ${atom.body.slice(0, 120)}${tags}${ref}`;
  }

  return `- ${atom.title} :: ${atom.body}${tags}${ref}`;
}

export function renderContext(atoms: ContextAtom[], query: string): string {
  const lines: string[] = [];
  lines.push("[CORTEXA_CONTEXT]");
  lines.push(`query: ${query}`);
  lines.push("");

  lines.push("[RELEVANT_ATOMS]");
  for (const atom of atoms) {
    lines.push(renderAtom(atom, "compact"));
  }

  return lines.join("\n");
}
```

### `packages/core/src/context/context-compiler.ts`

```ts
import { ContextAtom, CompiledContext } from "../types/context";
import { TokenCounter } from "../token/token-counter";
import { renderContext, renderAtom } from "./formatter";

export interface CompileOptions {
  model: string;
  maxTokens: number;
  reservedTokens?: number;
}

function scoreAtom(atom: ContextAtom): number {
  return (atom.priority * 0.45) + (atom.relevance * 0.4) + (atom.recency * 0.15);
}

function sortAtoms(atoms: ContextAtom[]): ContextAtom[] {
  return [...atoms].sort((a, b) => scoreAtom(b) - scoreAtom(a));
}

function tryPack(atoms: ContextAtom[], query: string, tokenCounter: TokenCounter, maxTokens: number): CompiledContext {
  const packed: ContextAtom[] = [];
  let rendered = renderContext([], query);
  let tokens = tokenCounter.countText(rendered);

  for (const atom of atoms) {
    const next = [...packed, atom];
    const candidate = renderContext(next, query);
    const candidateTokens = tokenCounter.countText(candidate);

    if (candidateTokens <= maxTokens) {
      packed.push(atom);
      rendered = candidate;
      tokens = candidateTokens;
      continue;
    }

    const compactAtom: ContextAtom = {
      ...atom,
      body: atom.body.slice(0, 180),
      title: atom.title.slice(0, 96)
    };

    const compactCandidate = renderContext([...packed, compactAtom], query);
    const compactTokens = tokenCounter.countText(compactCandidate);

    if (compactTokens <= maxTokens) {
      packed.push(compactAtom);
      rendered = compactCandidate;
      tokens = compactTokens;
      continue;
    }

    const symbolic = renderAtom(compactAtom, "symbolic");
    const symbolicCandidate = [
      "[CORTEXA_CONTEXT]",
      `query: ${query}`,
      "",
      "[RELEVANT_ATOMS]",
      ...packed.map((x) => renderAtom(x, "compact")),
      symbolic
    ].join("\n");

    const symbolicTokens = tokenCounter.countText(symbolicCandidate);

    if (symbolicTokens <= maxTokens) {
      packed.push({
        ...compactAtom,
        body: symbolic
      });
      rendered = symbolicCandidate;
      tokens = symbolicTokens;
    }
  }

  return { rendered, tokens, atoms: packed };
}

export function compileContext(query: string, atoms: ContextAtom[], options: CompileOptions): CompiledContext {
  const tokenCounter = new TokenCounter(options.model);
  const reserved = options.reservedTokens ?? 256;
  const effectiveBudget = Math.max(256, options.maxTokens - reserved);
  const ranked = sortAtoms(atoms);

  const compiled = tryPack(ranked, query, tokenCounter, effectiveBudget);
  const finalTokens = tokenCounter.countText(compiled.rendered);

  tokenCounter.free();

  return {
    rendered: compiled.rendered,
    tokens: finalTokens,
    atoms: compiled.atoms
  };
}
```

## 3) Custom embedding model for code semantics

For this part, I would not train from zero. I would build a **hybrid dual-encoder**:

* a code/text backbone, initialized from a pretrained code model
* a structural encoder that learns from AST features
* a projection head that maps everything into one shared semantic space

That gives you a model that understands both the surface text and the skeleton underneath.

### Embedding architecture

```text
Input code
  -> tokenizer
  -> transformer backbone
  -> pooled vector

AST features
  -> numeric feature extractor
  -> structural MLP

Fuse
  -> projection head
  -> L2 normalization
  -> embedding vector
```

```py
import math

FEATURE_NAMES = [

    "num_classes",
    "num_interfaces",
    "num_loops",
    "num_branches",
    "num_calls",
    "num_imports",

    "line_count",
    "identifier_count",
    "recursion_flag",
    "cyclomatic_proxy",
    "avg_identifier_len",
    "unique_dependency_count",
]

def build_feature_vector(entity_json: Dict) -> List[float]:
    """
    Turn one extracted entity or file summary into a fixed-length numeric vector.
    The vector is intentionally small so it is cheap to train on.
    """
    stats = entity_json.get("stats", {})
    dependencies = entity_json.get("dependencies", []) or []
    identifiers = entity_json.get("identifiers", []) or []

    line_count = float(stats.get("line_count", 0))
    avg_ident_len = 0.0
    if identifiers:
        avg_ident_len = sum(len(x) for x in identifiers) / float(len(identifiers))

    return [
        float(stats.get("num_functions", 0)),
        float(stats.get("num_classes", 0)),
        float(stats.get("num_interfaces", 0)),
        float(stats.get("num_loops", 0)),
        float(stats.get("num_branches", 0)),
        float(stats.get("num_calls", 0)),
        float(stats.get("num_imports", 0)),
        float(stats.get("num_returns", 0)),
        float(stats.get("max_depth", 0)),
        line_count,
        float(len(identifiers)),
        1.0 if stats.get("recursion_flag", False) else 0.0,
        float(stats.get("cyclomatic_proxy", 0)),
        avg_ident_len,
        float(len(set(dependencies))),
    ]

def pad_or_trim(vec: List[float], size: int = 15) -> List[float]:
    if len(vec) >= size:
        return vec[:size]
    return vec + [0.0] * (size - len(vec))
```

### `apps/ml/model.py`

```py
from __future__ import annotations
import torch
import torch.nn as nn
import torch.nn.functional as F
from transformers import AutoModel, AutoTokenizer

class StructuralEncoder(nn.Module):
    def __init__(self, in_dim: int = 15, hidden_dim: int = 128, out_dim: int = 128):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(in_dim, hidden_dim),
            nn.GELU(),
            nn.Dropout(0.1),
            nn.Linear(hidden_dim, out_dim),
            nn.GELU(),
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.net(x)

class CodeSemanticEmbedder(nn.Module):
    def __init__(
        self,
        backbone_name: str = "microsoft/codebert-base",
        struct_dim: int = 15,
        proj_dim: int = 256,
    ):
        super().__init__()
        self.tokenizer = AutoTokenizer.from_pretrained(backbone_name)
        self.backbone = AutoModel.from_pretrained(backbone_name)
        hidden = self.backbone.config.hidden_size

        self.struct_encoder = StructuralEncoder(struct_dim, 128, 128)

        self.fusion = nn.Sequential(
            nn.Linear(hidden + 128, 512),
            nn.GELU(),
            nn.Dropout(0.1),
            nn.Linear(512, proj_dim),
        )

    def encode_text(self, input_ids: torch.Tensor, attention_mask: torch.Tensor, struct_feats: torch.Tensor) -> torch.Tensor:
        out = self.backbone(input_ids=input_ids, attention_mask=attention_mask)
        cls_vec = out.last_hidden_state[:, 0, :]
        struct_vec = self.struct_encoder(struct_feats)
        fused = torch.cat([cls_vec, struct_vec], dim=-1)
        emb = self.fusion(fused)
        return F.normalize(emb, p=2, dim=-1)

    def forward(self, input_ids: torch.Tensor, attention_mask: torch.Tensor, struct_feats: torch.Tensor) -> torch.Tensor:
        return self.encode_text(input_ids, attention_mask, struct_feats)
```

### `apps/ml/train.py`

```py
from __future__ import annotations
import json
from dataclasses import dataclass
from typing import List, Dict

import torch
import torch.nn.functional as F
from torch.utils.data import Dataset, DataLoader
from transformers import AutoTokenizer
from model import CodeSemanticEmbedder
from feature_extractor import build_feature_vector, pad_or_trim

class PairDataset(Dataset):
    def __init__(self, path: str, tokenizer_name: str = "microsoft/codebert-base", max_len: int = 512):
        self.rows = [json.loads(line) for line in open(path, "r", encoding="utf-8") if line.strip()]
        self.tokenizer = AutoTokenizer.from_pretrained(tokenizer_name)
        self.max_len = max_len

    def __len__(self):
        return len(self.rows)

    def __getitem__(self, idx):
        row = self.rows[idx]
        code = row["code"]
        summary = row["summary"]
        feats = pad_or_trim(build_feature_vector(row.get("ast", {})))

        code_enc = self.tokenizer(
            code, truncation=True, padding="max_length", max_length=self.max_len, return_tensors="pt"
        )
        sum_enc = self.tokenizer(
            summary, truncation=True, padding="max_length", max_length=self.max_len, return_tensors="pt"
        )

        return {
            "code_input_ids": code_enc["input_ids"].squeeze(0),
            "code_attention_mask": code_enc["attention_mask"].squeeze(0),
            "sum_input_ids": sum_enc["input_ids"].squeeze(0),
            "sum_attention_mask": sum_enc["attention_mask"].squeeze(0),
            "struct_feats": torch.tensor(feats, dtype=torch.float32),
        }

def collate(batch):
    return {
        "code_input_ids": torch.stack([x["code_input_ids"] for x in batch]),
        "code_attention_mask": torch.stack([x["code_attention_mask"] for x in batch]),
        "sum_input_ids": torch.stack([x["sum_input_ids"] for x in batch]),
        "sum_attention_mask": torch.stack([x["sum_attention_mask"] for x in batch]),
        "struct_feats": torch.stack([x["struct_feats"] for x in batch]),
    }

def info_nce_loss(code_emb: torch.Tensor, text_emb: torch.Tensor, temperature: float = 0.07) -> torch.Tensor:
    logits = code_emb @ text_emb.t() / temperature
    labels = torch.arange(code_emb.size(0), device=code_emb.device)
    loss_a = F.cross_entropy(logits, labels)
    loss_b = F.cross_entropy(logits.t(), labels)
    return (loss_a + loss_b) * 0.5

def train(train_path: str, epochs: int = 3, batch_size: int = 8, lr: float = 2e-5, device: str = "cuda" if torch.cuda.is_available() else "cpu"):
    ds = PairDataset(train_path)
    dl = DataLoader(ds, batch_size=batch_size, shuffle=True, collate_fn=collate, drop_last=True)

    model = CodeSemanticEmbedder().to(device)
    optim = torch.optim.AdamW(model.parameters(), lr=lr)

    model.train()
    for epoch in range(epochs):
        total = 0.0
        for step, batch in enumerate(dl):
            for k in batch:
                batch[k] = batch[k].to(device)

            code_emb = model(batch["code_input_ids"], batch["code_attention_mask"], batch["struct_feats"])
            zero_struct = torch.zeros_like(batch["struct_feats"])
            text_emb = model(batch["sum_input_ids"], batch["sum_attention_mask"], zero_struct)

            loss = info_nce_loss(code_emb, text_emb)

            optim.zero_grad()
            loss.backward()
            torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
            optim.step()

            total += float(loss.item())
            if step % 20 == 0:
                print(f"epoch={epoch} step={step} loss={loss.item():.4f}")

        print(f"epoch={epoch} avg_loss={total / max(1, len(dl)):.4f}")

    torch.save(model.state_dict(), "data/checkpoints/code_semantic_embedder.pt")

if __name__ == "__main__":
    train("data/training/pairs.jsonl")
```

### What this model learns

It learns two channels at once:

* **surface semantics** from the transformer backbone
* **structural semantics** from AST-derived statistics

That means it can separate code that looks similar from code that actually behaves similarly. That is the difference between a parrot and a navigator.

## 4) Multi-agent memory evolution system

This is the part that makes your memory layer feel alive instead of archival.

The loop is:

1. **Writer** proposes a candidate memory atom from raw input
2. **Critic** scores novelty, quality, redundancy, and usefulness
3. **Consolidator** merges duplicates into a canonical memory
4. **Archivist** applies decay, promotion, and pruning rules
5. **Store** writes the evolved result back to Mempalace

```ts
import { MemoryAtom } from "../../types/memory";

export class ArchivistAgent {
  applyDecay(atom: MemoryAtom, now = Date.now()): MemoryAtom {
    const ageDays = Math.max(0, (now - atom.lastAccessedAt) / (1000 * 60 * 60 * 24));
    const decay = Math.exp(-ageDays / 30);

    return {
      ...atom,
      importance: Math.max(0, atom.importance * decay),
      confidence: Math.max(0, atom.confidence * decay * 0.98)
    };
  }

  shouldPromote(atom: MemoryAtom): boolean {
    return atom.importance >= 0.75 && atom.confidence >= 0.65;
  }

  shouldArchive(atom: MemoryAtom): boolean {
    return atom.importance < 0.15 && atom.confidence < 0.25;
  }
}
```

### `packages/core/src/memory/evolution/engine.ts`

```ts
import { WriterAgent } from "./writer.agent";
import { CriticAgent } from "./critic.agent";
import { ConsolidatorAgent } from "./consolidator.agent";
import { ArchivistAgent } from "./archivist.agent";
import { MemoryAtom } from "../../types/memory";

export interface MemoryStore {
  searchSimilar(text: string, topK: number): Promise<MemoryAtom[]>;
  upsert(atom: MemoryAtom): Promise<void>;
  archive(atomId: string): Promise<void>;
}

export class MemoryEvolutionEngine {
  private archivist = new ArchivistAgent();

  constructor(
    private writer: WriterAgent,
    private critic: CriticAgent,
    private consolidator: ConsolidatorAgent,
    private store: MemoryStore
  ) {}

  async evolve(input: {
    projectId: string;
    sourceType: "code" | "chat" | "manual";
    rawText: string;
    context?: string;
  }): Promise<{ stored: boolean; action: string; reason: string }> {
    const writerOut = await this.writer.propose(input);

    let stored = false;
    let lastAction = "none";
    let lastReason = "no candidate";

    for (const candidate of writerOut.candidates) {
      const neighbors = await this.store.searchSimilar(candidate.summary, 5);

      const review = await this.critic.review({
        title: candidate.title,
        summary: candidate.summary,
        content: candidate.content,
        existingSnippets: neighbors.map((n) => n.summary)
      });

      lastAction = review.action;
      lastReason = review.reason;

      if (!review.accepted || review.action === "reject") {
        continue;
      }

      let finalAtom: MemoryAtom = {
        id: cryptoRandomId(),
        kind: candidate.kind,
        projectId: input.projectId,
        sourceType: input.sourceType,
        title: candidate.title,
        summary: candidate.summary,
        content: candidate.content,
        tags: candidate.tags ?? [],
        importance: review.score,
        confidence: Math.max(0.1, review.clarity),
        createdAt: Date.now(),
        lastAccessedAt: Date.now(),
        sourceRef: candidate.sourceRef
      } as MemoryAtom;

      if (review.action === "merge" && neighbors.length > 0) {
        const merged = await this.consolidator.merge({
          candidate: finalAtom,
          neighbors
        });

        finalAtom = {
          ...finalAtom,
          title: merged.title,
          summary: merged.summary,
          content: merged.content,
          tags: merged.tags,
          confidence: merged.confidence
        };
      }

      finalAtom = this.archivist.applyDecay(finalAtom);

      if (this.archivist.shouldArchive(finalAtom)) {
        await this.store.archive(finalAtom.id);
        continue;
      }

      await this.store.upsert(finalAtom);
      stored = true;
    }

    return { stored, action: lastAction, reason: lastReason };
  }
}

function cryptoRandomId(): string {
  return Math.random().toString(36).slice(2, 12) + Date.now().toString(36);
}
```

## 5) Detailed build procedure

### Phase A, make the extractor real

1. install Tree-sitter and the first grammars
2. point the extractor at a workspace folder
3. extract entities into JSONL
4. verify function, class, loop, branch extraction on a small repo

### Phase B, wire exact budgeting

1. serialize a context prompt
2. count the final string with `TokenCounter`
3. shrink or drop atoms until under budget
4. compare output against your target model’s limit

### Phase C, train semantic embeddings

1. generate training pairs from:

   * function body ↔ summary
   * code chunk ↔ chat resolution
   * duplicate solutions across projects
2. train the hybrid encoder
3. store embeddings in your vector index
4. retrieve by semantic similarity first, then rerank with importance and recency

### Phase D, activate memory evolution

1. every ingest event goes to writer
2. critic filters low-value noise
3. consolidator merges similar items
4. archivist decays stale memory
5. promoted memories become part of the global pool

1. `extractor.ts`

```ts
export interface MemoryRow {
  id: string;
  kind: string;
  projectId: string;
  title: string;
  summary: string;
  content: string;
  importance: number;
  confidence: number;
  createdAt: number;
  lastAccessedAt: number;
}

export class MemoryStore {
  insert(mem: MemoryRow) {
    const stmt = db.prepare(`
      INSERT OR REPLACE INTO memories
      VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
    `);

    stmt.run(
      mem.id,
      mem.kind,
      mem.projectId,
      mem.title,
      mem.summary,
      mem.content,
      mem.importance,
      mem.confidence,
      mem.createdAt,
      mem.lastAccessedAt
    );
  }

  search(query: string) {
    const stmt = db.prepare(`
      SELECT * FROM memories
      WHERE summary LIKE ?
      ORDER BY importance DESC
      LIMIT 10
    `);

    return stmt.all(`%${query}%`);
  }

  all() {
    return db.prepare(`SELECT * FROM memories`).all();
  }
}
```

```ts
import { Command } from "commander";
import fs from "fs";
import path from "path";
import { extractEntitiesFromFile } from "@cortexa/core";
import { MemoryStore } from "@cortexa/core";

export const ingestCommand = new Command("ingest")
  .argument("<dir>", "project directory")
  .action((dir) => {
    const store = new MemoryStore();

    function walk(folder: string) {
      const files = fs.readdirSync(folder);

      for (const file of files) {
        const full = path.join(folder, file);

        if (fs.statSync(full).isDirectory()) {
          walk(full);
        } else if (file.endsWith(".ts") || file.endsWith(".cpp")) {
          const result = extractEntitiesFromFile(full);

          result.entities.forEach((e) => {
            store.insert({
              id: e.id,
              kind: e.kind,
              projectId: dir,
              title: e.name,
              summary: e.summary,
              content: e.rawSlice || "",
              importance: 0.5,
              confidence: 0.5,
              createdAt: Date.now(),
              lastAccessedAt: Date.now()
            });
          });
        }
      }
    }

    walk(dir);

    console.log("Ingestion complete.");
  });
```

```ts
import fs from "fs";
import path from "path";

export function loadCopilotChats(projectPath: string) {
  const base = path.join(projectPath, ".vscode", "workspaceStorage");

  if (!fs.existsSync(base)) return [];

  const sessions: any[] = [];

  for (const folder of fs.readdirSync(base)) {
    const chatPath = path.join(base, folder, "chatSessions");

    if (!fs.existsSync(chatPath)) continue;

    for (const file of fs.readdirSync(chatPath)) {
      const full = path.join(chatPath, file);
      const data = JSON.parse(fs.readFileSync(full, "utf8"));

      sessions.push(...parseSession(data));
    }
  }

  return sessions;
}

function parseSession(raw: any) {
  const out = [];

  for (const msg of raw.messages || []) {
    out.push({
      prompt: msg.prompt || "",
      response: msg.response || "",
      timestamp: msg.timestamp
    });
  }

  return out;
}
```

## 1.5 Graph Service

```ts
// packages/core/src/graph/graph.service.ts
import { v4 as uuid } from "uuid";

export class GraphService {
  addNode(node: any) {
    db.prepare(`
      INSERT OR REPLACE INTO nodes VALUES (?, ?, ?, ?)
    `).run(
      node.id,
      node.type,
      node.label,
      JSON.stringify(node.metadata || {})
    );
  }

  addEdge(edge: any) {
    db.prepare(`
      INSERT INTO edges VALUES (?, ?, ?, ?, ?, ?)
    `).run(
      uuid(),
      edge.from,
      edge.to,
      edge.type,
      edge.weight || 1.0,
      JSON.stringify(edge.metadata || {})
    );
  }

  getNeighbors(nodeId: string) {
    return db.prepare(`
      SELECT * FROM edges WHERE fromNode = ? OR toNode = ?
    `).all(nodeId, nodeId);
  }
}
```

```ts
const wss = new WebSocket.Server({ port: 4321 });
const store = new HybridMemoryStore();

wss.on("connection", (ws) => {
  ws.on("message", async (msg) => {
    const { query, step } = JSON.parse(msg.toString());

    // Step-aware retrieval
    const memories = await store.search(query);

    const atoms = memories.map((m: any) => ({
      id: m.id,
      kind: "memory",
      title: m.title,
      body: m.summary,
      priority: m.importance,
      recency: 0.5,
      relevance: 0.9
    }));

    const ctx = compileContext(query, atoms, {
      model: "gpt-4o-mini",
      maxTokens: 2000
    });

    ws.send(JSON.stringify({
      type: "context_update",
      step,
      context: ctx.rendered
    }));
  });
});
```

# CORTEXA Complete Blueprint

## 1. Project Identity

**Project name:** CORTEXA

**Universal bridge layer:** CX-LINK

**Purpose:**
A local-first, reusable developer intelligence system that ingests code, chat, project structure, ASTs, graphs, embeddings, and refactoring signals, then delivers compact, high-signal context to any agent, editor, or CLI through a universal adapter layer.

---

## 2. Core Vision

CORTEXA is not a single app. It is a **memory and reasoning substrate** for software work.

It should:

* remember across projects
* understand code structurally, not just textually
* compress context into token-efficient formats
* stream evolving context in real time
* coordinate multiple agents over the same memory fabric
* expose everything through a universal interface that works with GitHub Copilot, Copilot CLI, VS Code, terminal agents, and future editor integrations

---

## 3. Design Principles

1. **Local-first**

   * All sensitive project knowledge stays on the user’s machine unless explicitly exported.

2. **Semantic over lexical**

   * Raw code is not the final product. The system stores structured meaning.

3. **Incremental over full reprocessing**

   * File changes should update only the affected AST nodes, graph edges, and memory atoms.

4. **Token-aware by design**

   * Every retrieval and context package must be measured against the target model’s token budget.

5. **Universal delivery**

   * The same backend should feed multiple agents with adapters rather than forks.

6. **Observable and debuggable**

   * Every memory atom, graph edge, and injection event should be traceable.

7. **Composable intelligence**

   * Retrieval, graph reasoning, refactoring, and streaming should be independent modules.

---

## 10. AST Extraction Blueprint

### Goals

* parse files using Tree-sitter
* extract functions, classes, methods, loops, branches, imports
* infer rough complexity hints
* emit code entities and graph nodes
* generate compact summaries

### Incremental Update Strategy

* watch file changes
* compute file hash before and after
* if unchanged, skip
* if changed, reparse only that file
* replace affected entities and graph edges
* preserve unchanged neighboring nodes and global memory state

### Output of extraction

Each parsed unit should produce:

* entity id
* kind
* name
* signature
* source range
* summary
* complexity hint
* dependencies
* source hash
* graph links

* entity summary

* chat turn summary
* refactor plan summary
* concept memory summary

### Embedding architecture

A hybrid encoder with:

* pretrained code-text transformer backbone
* AST structural feature encoder
* projection head into shared vector space

### Pipeline

```text
Raw text + AST stats
  ↓
Tokenizer + structural feature extractor
  ↓
Backbone encoder
  ↓
Fusion head
  ↓
Normalized embedding
  ↓
Vector DB upsert
```

### Training objective

Use contrastive learning so that:

* code and its summary are close
* duplicate patterns from different projects are close
* unrelated code is far apart

### Graph-aware retrieval

* semantic similarity
* graph centrality

## 13. Context Compiler Blueprint

### Inputs

* user query
* current file scope
* project memory atoms
* graph neighbors
* relevant chat history
* active refactor plan

### Process

1. retrieve seed candidates via vector search
2. expand with graph neighbors
3. rerank by importance and recency
4. compress long bodies into summaries
5. pack until exact token budget is reached
6. output structured context format

### Output format

Use a strict context envelope with sections:

* intent
* scope
* concepts
* code entities
* graph facts
* relevant memory
* history
* constraints

### Token counting rule

Count the final rendered text, not an abstract estimate.
The compiler must stop once the exact token budget is met or exceeded and then back off to the last valid pack.

## 20. Security and Isolation Blueprint

### Must-have rules

* project namespaces are isolated by default
* cross-project memory sharing requires explicit global memory tagging
* local database access must be gated by a daemon or CLI token
* streaming sessions should be scoped to a project and user session
* graph edges and embeddings should not leak unrelated workspace data

### Sensitive handling

* do not mirror secrets into memory
* skip `.env`, keys, tokens, certificates, and sensitive files
* allow exclude lists at project level

* incremental parse only
* stream deltas, not full snapshots

### Large project strategy

* bound graph traversal depth
* cap expansion fanout
* compress stale memories
